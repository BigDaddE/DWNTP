        finishingTps: 10000
    workload:
      module: workload/logEvent.js

    description: >-
      Maintain a constant backlog of 500 unconfirmed transactions to test queue
      saturation.
    txNumber: 10000
    rateControl:
      type: fixed-load
      opts:
        transactionLoad: 500
        startingTps: 100
    workload:
      module: workload/logEvent.js
  - label: Query-Performance
    description: Submit read-only GetAllEvents queries at 1000 TPS for 60 seconds.

    rateControl:
      type: fixed-rate

        tps: 1000
